class Car
{
	String brand;
	int year;
	
	void displayinfo()
	{
		System.out.println("Brand: " + brand);
	System.out.println("Year: " + year);
	}
}

public class ClassExample
{
	public static void main(String[] args)
	{
		Car myCar = new Car();
	
		myCar.brand = "Toyota";
		myCar.year = 2022;
		
		myCar.displayinfo();
	}
}
