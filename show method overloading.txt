class MethodOperation
{
	int add(int a)
	{
		return a;
	}
	int add(int a, int b)
	{
		return a + b;
	}
	
	double add(double a, double b)
	{
		return a + b;
	}
}
public class OverlodingExample
{
	public static void main(String[] args)
	{
		MethodOperation mo = new MethodOperation();
		System.out.println("Sum with one argument:" + mo.add(10));
		System.out.println("Sum with two arguments: " + mo.add(10,5));
		System.out.println("Sum with two double arguments: " + mo.add(10.5,5.5));
	}
}
