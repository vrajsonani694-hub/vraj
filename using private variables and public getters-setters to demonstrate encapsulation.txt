class Employee
{
	private String name;

	public String getName()
	{
		return name;
	}
	public void setName(String newName)
	{
		this.name = newName;
	}
}
public class Encapsulation
{
	public static void main(String[] args)
	{
		Employee emp = new Employee();
		emp.setName("Tushar");
		System.out.println("Employee Name: " + emp.getName());
	}
}
